Misc Scripts that I use.

mp3 to libopus for audiobooks

for f in *.mp3; do ffmpeg -i "${f}" -c:a libopus -b:a 32k "${f%%.*}.ogg"; done

ffmpeg change container of file

ffmpeg -i input.webm -acodec copy output.ogg              # webm -> ogg (containing opus from yt-dlp)
ffmpeg -i input.mp4 -acodec copy -vcodec copy output.webm # any container to webm/mkv

yt-dlp playlist with order in filename

yt-dlp -f "bestvideo[height<=1080]+bestaudio" -o "%(playlist_index)s-%(title)s.%(ext)s"
